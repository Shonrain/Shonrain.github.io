  ids.value = ids;

# API

# Props

属性说明类型默认值
ids选中项的idArray<number>[]
visible是否显示选择框booleanfalse
max最大可选数量（多选时生效）number-
selectType选项类型（radio 或者 checkbox）stringcheckbox

# 事件

事件名称类型说明回调参数说明
ok(ids: number[]) => void点击确认选中的券包分类 id 的集合
cancel() => void点击取消-
change(value: CouponpacketCategory[]) => void选中项发生改变时触发选中的券包分类

# 类型

# interface CouponpacketCategory

export interface CouponpacketCategory {
  id: number, // 分类ID
  packetCategoryName: string, // 分类名称
  packetCategoryType: CARDPACKET_CATEGORY_TYPE, // 上架渠道
  sort:number, // 排序
  num:number, // 券包中券数量
  status: CARDPACKET_CATEGORY_STATUS, // 状态
}

# enum CARDPACKET_CATEGORY_TYPE、CARDPACKET_CATEGORY_STATUS

// 券包分类上架渠道
export enum CARDPACKET_CATEGORY_TYPE {
  WX=1, // 微信
  ALI, // 支付宝
  ALL=''// 全部'
}
// 状态
export enum CARDPACKET_CATEGORY_STATUS {
  OFF, // 禁用
  ON, // 开启
  ALL=''// 全部'
}
