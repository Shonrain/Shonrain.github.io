# ChooseLink 选择链接

本页的例子都需要设置token，如果提示错误请登录dev环境获取token

# 用法示例

# 基本用法

选择跳转链接（默认不跳转）
查看代码
<template>
  <div class="example-item">
    <choose-link v-model:link="link" />
  </div>
</template>

<script setup lang="ts">
import { ChooseLink } from 'miya-design';
import { Link, newLink } from 'miya-design/types';

const link = ref<Link>(newLink());
</script>

# 调整左侧 menu 的个数

选择跳转链接（默认不跳转）
查看代码
<template>
  <div class="example-item">
    <choose-link
      v-model:link="link" 
      :menus="menus" 
    />
  </div>
</template>

<script setup lang="ts">
import { ChooseLink } from 'miya-design';
import { Link, newLink, LINK_TYPE } from 'miya-design/types';

const link = ref<Link>(newLink());
const menus = [LINK_TYPE.COMMON_ENTRANCE, LINK_TYPE.GOODS_DETAIL, LINK_TYPE.STORE, LINK_TYPE.GIFT_CARD_DETAIL];
</script>

# 直接通过实例获取链接在C端的路径

import { Link, newLink } from 'miya-design/types';

const link: Link = newLink();

// 不带参数
link.getPath();

// 带参数
link.getPath({foo: 1, bar: 2})

# 类型

# class Link
查看完整代码
export class Link {
  [key: string]: any;

  type: TYPE;// 链接类型

  entrance?: COMMON_ENTRANCE; // 常用入口

  path?: string; // 路径

  params: LinkParams; // 链接参数

  // 省略了实现细节，获取该链接在C端的路径，支持传参
  getPath: (params: Record<string, any> = {}) => string
}

// 省略了实现细节，实例化一个链接
export const newLink = (entrance?: COMMON_ENTRANCE): Link => Link
# interface LinkParams
查看完整代码
export interface LinkParams {
  [key: string]: number | string | undefined,
  id?: string | number; // 某些链接需要此参数
  appId?: string | number; // 小程序链接需要此参数
  name?: string; // 链接的名称，用于回显
  extraData?: string; // 额外参数，主要用于小程序链接
  plugin?: string; // 支付宝插件id
  pluginName?: string; // 支付宝插件名称
}
