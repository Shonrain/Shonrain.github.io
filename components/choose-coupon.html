# ChooseCoupon 选择优惠券

本页的例子都需要设置token，如果提示错误请登录dev环境获取token

# 基本用法

点击查看代码
<template>
  <div class="example-item">
    <a-button type="primary" @click="visible = true">选择优惠券</a-button>
    <div class="choose-coupon">
      <choose-coupon :visible="visible" @ok="ok" @cancel="visible = false" :ids="ids" />
    </div>
  </div>
</template>

<script setup lang="ts">
import { ChooseCoupon } from 'miya-design';

const visible = ref(false);
const ids = ref([]);
const ok = (ids: number[]) => {
  ids.value = ids;
}
</script>

# API

# Props

属性说明类型默认值
ids选中项的idArray<number>[]
visible是否显示选择框booleanfalse
max最大可选数量（多选时生效）number-
selectType选项类型（radio 或者 checkbox）stringcheckbox

# 事件

事件名称类型说明回调参数说明
ok(ids: number[]) => void点击确认选中优惠券 id 的集合
cancel() => void点击取消-
change(value: Coupon[]) => void选中项发生改变时触发选中的优惠券（不支持翻页）

# 类型

# interface Coupon
export interface Coupon {
  id: number | string,
  cardTitle: string,
  couponType: COUPON_TYPE,
  status: boolean | COUPON_STATUS,
  couponStockId: string | null,
  cardType?: CARD_TYPE,
  createTime: string,
  endTimestamp?: string,
  beginTimestamp?: string,
  isSupportGiven: COUPON_SUPPORT_GIVEN
}
# enum COUPON_TYPE
export enum COUPON_TYPE {
  FREE_COUPON, // 免购券
  VALUE_COUPON // 预购券
}
# enum CARD_TYPE
export enum CARD_TYPE {
  WX = 'WX', // 微信券
  MIYA = 'MIYA', // 米雅券
  ALI = 'ALI', // 支付宝券
  MCH_CRM = 'MCH_CRM' // 商户券
}
# enum COUPON_STATUS
export enum COUPON_STATUS {
  OFF,
  ON
}
# enum COUPON_SUPPORT_GIVEN
export enum COUPON_SUPPORT_GIVEN {
  NO,
  YES,
}
