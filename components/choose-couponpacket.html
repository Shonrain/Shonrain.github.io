# API

# Props

属性说明类型默认值
ids选中项的idArray<number>[]
visible是否显示选择框booleanfalse
max最大可选数量（多选时生效）number-
selectType选项类型（radio 或者 checkbox）stringcheckbox

# 事件

事件名称类型说明回调参数说明
ok(ids: number[]) => void点击确认选中的券包 id 的集合
cancel() => void点击取消-
change(value: Couponpacket[]) => void选中项发生改变时触发选中的券包（不支持翻页）

# 类型

# interface Couponpacket

export interface Couponpacket {
  id: number, // ID
  cardPacketName: string, // 券包名称
  cardPacketType: CARDPACKET_TYPE, // 券包类型
  cardPacketPrice:number, // 售价
  sort:number, // 排序
  status: PACKET_STATUS, // 上架状态
  saleStatus: PACKET_SALE_STATUS, // 可售状态
  stock:number, // 库存
  saleStartTime:string, // 开售时间
  saleEndTime:string, // 停售时间
  createTime:string, // 创建时间
}

# enum CARDPACKET_TYPE、PACKET_STATUS、PACKET_SALE_STATUS

// 券包类型
export enum CARDPACKET_TYPE {
  MIYA = 0, // 米雅券
  WX, // 微信券
  ALI, // 支付宝券
  MCH_CRM // 商户券
}
// 上架状态
export enum PACKET_STATUS {
  DOWN, // 下架
  UP, // 上架
  PREPARE, // 预上架
}
// 可售状态
export enum PACKET_SALE_STATUS {
  STOPPED = 1, // 已停售
  NO_SALE, // 未开售
  ON_SALE, // 售卖中
}
