# ChooseLoyaltyCard 选择集点卡

本页的例子都需要设置token，如果提示错误请登录dev环境获取token

# 基本用法

点击查看代码
<template>
  <div class="example-item">
    <a-button type="primary" @click="visible = true">选择集点卡</a-button>
    <div class="choose-loyalty-card">
      <choose-loyalty-card :visible="visible" @ok="ok" @cancel="visible = false" :ids="ids" />
    </div>
  </div>
</template>

<script setup lang="ts">
import { ChooseLoyaltyCard } from 'miya-design';

const visible = ref(false);
const ids = ref([]);
const ok = (ids: number[]) => {
  ids.value = ids;
}
</script>

# API

# Props

属性说明类型默认值
ids选中项的idArray<number>[]
visible是否显示选择框booleanfalse
max最大可选数量（多选时生效）number-
selectType选项类型（radio 或者 checkbox）stringcheckbox

# 事件

事件名称类型说明回调参数说明
ok(ids: number[]) => void点击确认选中选择集点卡 id 的集合
cancel() => void点击取消-
change(value: LoyaltyCard[]) => void选中项发生改变时触发选中的选择集点卡

# 类型

# interface LoyaltyCard
export interface LoyaltyCard {
  id: number,
  title: string,
  onChannel: LOYALTY_CARD_CHANNEL,
  activityStatus: string, // NOSTART：未开始，ON：进行中，OFF：已结束
  status: string, // ON: 上架，OFF：下架
  setpointStartTime: string,
  setpointEndTime: string,
  exchangeStartTime: string,
  exchangeEndTime: string,
}
# enum LOYALTY_CARD_CHANNEL
export enum LOYALTY_CARD_CHANNEL {
  WECHAT = 'WX',
  ALIPAY = 'ZFB'
}
