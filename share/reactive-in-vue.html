referentially transparent and pure functions

An expression is said to be referentially transparent if it can be replaced with its corresponding value without changing the program’s behavior. As a result, evaluating a referentially transparent function gives the same value for same arguments. Such functions are called pure functions.

const add(x: number, y: number) => x + y;

console.log(add(1, 2)) // 3
console.log(add(1, 2)) // 3

let activeEffect: Function | null = null;

const bucket: WeakMap<any, Map<string, Set<Function>>> = new WeakMap();

const data = { a: 1 };

const obj = new Proxy(data, {
  get(target: any, key: string) {
    if (!activeEffect) return target[key];
    let depsMap = bucket.get(target);
    if (!depsMap) {
      depsMap = new Map()
      bucket.set(target, depsMap);
    }
    let deps = depsMap.get(key);
    if (!deps) {
      deps = new Set();
      depsMap.set(key, deps);
    }
    deps.add(activeEffect);
    return target[key];
  },
  set(target: any, key: string, value: any) {
    target[key] = value;
    const depsMap = bucket.get(target);
    if (depsMap) {
      const deps = depsMap.get(key);
      deps && deps.forEach(fn => fn());
    }
    return true;
  }
});

let activeEffect: Function | null = null;

const bucket: WeakMap<any, Map<string, Set<Function>>> = new WeakMap();

const track = (target: any, key: string) => {
  if (!activeEffect) return;
  let depsMap = bucket.get(target);
  if (!depsMap) {
    depsMap = new Map()
    bucket.set(target, depsMap);
  }
  let deps = depsMap.get(key);
  if (!deps) {
    deps = new Set();
    depsMap.set(key, deps);
  }
  deps.add(activeEffect);
}

const trigger = (target: any, key: string) => {
  const depsMap = bucket.get(target);
  if (depsMap) {
    const effects = depsMap.get(key);
    effects && effects.forEach(fn => fn());
  }
}

const reactive = (data: any) => {
  return new Proxy(data, {
    get(target: any, key: string) {
      track(target, key);
      return target[key];
    },
    set(target: any, key: string, value: any) {
      target[key] = value;
      trigger(target, key);
      return true;
    }
  });
}

type EffectFunction = {
  (): void,
  deps: Array<Set<EffectFunction>>
}

let activeEffect: EffectFunction | null = null;
const bucket: WeakMap<any, Map<string, Set<EffectFunction>>> = new WeakMap();
const cleanup = (effectFn: EffectFunction) => {
  effectFn.deps.forEach(item => item.delete(effectFn));
  effectFn.deps.length = 0;
}
const effect = (fn: Function) => {
  const effectFn = () => {
    cleanup(effectFn);
    activeEffect = effectFn;
    fn();
  }
  effectFn.deps = [] as Array<Set<EffectFunction>>;
  effectFn();
}

type EffectFunction = {
  (): void,
  deps: Array<Set<EffectFunction>>
}

let activeEffect: EffectFunction | null = null;
const effectStack: Array<EffectFunction> = [];
const bucket: WeakMap<any, Map<string, Set<EffectFunction>>> = new WeakMap();

export const effect = (fn: Function) => {
  const effectFn = () => {
    cleanup(effectFn);
    activeEffect = effectFn;
    effectStack.push(effectFn);
    fn();
    effectStack.pop();
    activeEffect = effectStack[effectStack.length - 1]
  }
  effectFn.deps = [] as Array<Set<EffectFunction>>;
  effectFn();
}
